    console.time('Hello')

Declaration: Var, Let, Const

  • Var : variable declaration
  • Let : Similar to Var
  • Const: cannot reassign, but object's property can be changed.
    const msg = "Hello world";
    Msg = "Welcome";  // not accepted
    
    const person = { name: "ABC", age: 30 };
    Person.age = 35;   // accepted

    Const num = [1,2,3,4,5];
    Num.push(6);   // accepted

To Number:

    a = Number('5')
    a.toFixed(2); // 5.00

    b = Number(true);   //1
    b = Number(false);  //0
    b = Number(null);    //0
    b = Number('hello');      //NaN
    b = Number([1,2,3,4]);  //NaN

    b = parseInt('100');          //100
    b = parseFloat('100.30');  //100.30

Template strings

ES5:

            '<li>City: ' + city + '</li>' +
    '</ul>';

ES6:

    Html = `<ul> 
            <li>Name: ${name} </li> 
            <li>Age: ${age} </li>

            <li>City: ${4+6} </li>
